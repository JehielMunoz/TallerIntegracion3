	SELECT Rut, Nombre, F_nacimiento, F_ingreso, Direccion, Comuna, Curso, 	Curso_anterior, Establecimiento_ant, Activo, Repitente, Ingles, Estado, 	Repitio FROM `talumno` 	WHERE Nombre Like '%El%'

	SELECT Rut, Nombre, F_nacimiento, F_ingreso, Direccion, Comuna, Curso, 	Curso_anterior, Establecimiento_ant, Activo, Repitente, Ingles, Estado, 	Repitio FROM `talumno`

-muestra datos de contacto
	SELECT rap.Parentesco,tp.Fono, tp.Email FROM tpadres tp 
	JOIN rel_talumno_tpadres rap on tp.Rut= rap.Rut_padre
	JOIN talumno ta on rap.Rut_alu = ta.Rut
	WHERE
	ta.Rut = '199586349'
